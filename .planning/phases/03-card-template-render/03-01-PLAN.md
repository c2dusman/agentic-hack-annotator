**Function: `escapeHtml(str)`**
- Convert `str` to String, then replace `&` with `&amp;`, `<` with `&lt;`, `>` with `&gt;`, `"` with `&quot;`
- Defensive against unexpected AI-generated content containing HTML chars

**Function: `buildStepsHtml(steps)`**
- Takes array of `{ number, label, description }` objects
- Returns HTML string with one `<div class="step">` per step
- Each step contains: `<div class="step-number">${step.number}</div>` and `<div class="step-content"><span class="step-label">${escapeHtml(step.label)}</span><span class="step-description">${escapeHtml(step.description)}</span></div>`
- Join with `\n`

**Function: `populateTemplate(template, annotationData, screenshotBase64, focus, pageUrl)`**
- Chain `.replace()` calls (not `.replaceAll()`) for all 7 placeholders
- Order: `{{CARD_TITLE}}` first (escaped), `{{CARD_SUBTITLE}}` (escaped), `{{STEPS_HTML}}` (already escaped internally by buildStepsHtml), `{{PAGE_URL}}` (escaped), `{{FOCUS_LABEL}}` (escaped, empty string if focus is null), `{{FOCUS_BADGE_STYLE}}` (empty string if focus is truthy, `'display:none'` if null/falsy), `{{SCREENSHOT_BASE64}}` LAST (largest substitution per RESEARCH.md pitfall 3)

**Function: `getLaunchOptions()`** — keep exactly as-is from current stub:
```javascript
function getLaunchOptions() {
  return {
    headless: true,
    args: ['--no-sandbox', '--disable-setuid-sandbox', '--disable-dev-shm-usage', '--disable-gpu']
  };
}
```

**Function: `async renderCard(annotationData, screenshotBase64, focus = null, pageUrl = '')`**
- Declare `let browser = null; let page = null;`
- In try block:
  1. Read template: `const template = fs.readFileSync(path.join(__dirname, '../templates/card.html'), 'utf8');`
  2. Populate: `const html = populateTemplate(template, annotationData, screenshotBase64, focus, pageUrl);`
  3. Launch browser: `browser = await puppeteer.launch(getLaunchOptions());`
  4. Create page: `page = await browser.newPage();`
  5. Set viewport: `await page.setViewport({ width: 1080, height: 1920 });`
  6. Set content: `await page.setContent(html, { waitUntil: 'domcontentloaded' });` (NOT networkidle0 — per D-04 and RESEARCH.md pitfall 1)
  7. Wait for fonts: `await page.evaluate(() => document.fonts.ready);` (per D-04)
  8. Screenshot: `const rawBuffer = await page.screenshot({ type: 'png', clip: { x: 0, y: 0, width: 1080, height: 1920 } });`
  9. Compress with Sharp: `const optimized = await sharp(rawBuffer).png({ compressionLevel: 9, palette: false }).toBuffer();` (per D-07)
  10. Write output: `ensureOutputDir(); const filename = generateId() + '.png'; const filepath = path.join(process.env.OUTPUT_DIR || './output', filename); fs.writeFileSync(filepath, optimized);`
  11. Return: `{ filename, filepath }`
- In finally block:
  1. `if (page) await page.close().catch(() => {});`
  2. `if (browser) await browser.close().catch(() => {});`

**Module exports:** `module.exports = { renderCard, getLaunchOptions };`

Do NOT export `escapeHtml`, `buildStepsHtml`, or `populateTemplate` — they are internal helpers.

**Test script structure:**
```javascript
'use strict';
require('dotenv').config();
const { renderCard } = require('../src/render');
const sharp = require('sharp');
const fs = require('fs');
const path = require('path');

// Mock annotation data matching annotate.js output structure
const mockAnnotation = {
  cardTitle: 'Connect Your Gmail Account',
  cardSubtitle: 'Step-by-step guide to linking Gmail with the dashboard for instant notifications',
  steps: [
    { number: 1, label: 'Open Settings', description: 'Navigate to the Settings page from the main dashboard sidebar menu' },
    { number: 2, label: 'Select Integrations', description: 'Click the Integrations tab to see available third-party connections' },
    { number: 3, label: 'Authorize Gmail', description: 'Click Connect next to Gmail and sign in with your Google account credentials' },
    { number: 4, label: 'Confirm Access', description: 'Review the permissions requested and click Allow to complete the connection' }
  ]
};

// Create a small 1200x800 red PNG as mock screenshot (base64)
async function createMockScreenshot() {
  const buf = await sharp({
    create: { width: 1200, height: 800, channels: 3, background: { r: 30, g: 60, b: 120 } }
  }).png().toBuffer();
  return buf.toString('base64');
}

async function runTest() {
  console.log('Creating mock screenshot...');
  const screenshotBase64 = await createMockScreenshot();

  // Test 1: Render with focus hint
  console.log('Test 1: renderCard with focus hint...');
  const result1 = await renderCard(mockAnnotation, screenshotBase64, 'How to connect Gmail', 'https://example.com/settings');
  console.log('  Output:', result1.filename);

  // Verify file exists
  if (!fs.existsSync(result1.filepath)) throw new Error('Output file not created');

  // Verify dimensions via sharp metadata
  const meta1 = await sharp(result1.filepath).metadata();
  if (meta1.width !== 1080) throw new Error(`Width is ${meta1.width}, expected 1080`);
  if (meta1.height !== 1920) throw new Error(`Height is ${meta1.height}, expected 1920`);
  if (meta1.format !== 'png') throw new Error(`Format is ${meta1.format}, expected png`);
  console.log('  PASS: 1080x1920 PNG with focus');

  // Test 2: Render without focus hint (badge hidden)
  console.log('Test 2: renderCard without focus hint...');
  const result2 = await renderCard(mockAnnotation, screenshotBase64, null, 'https://example.com/settings');
  const meta2 = await sharp(result2.filepath).metadata();
  if (meta2.width !== 1080 || meta2.height !== 1920) throw new Error('Dimensions wrong for no-focus render');
  console.log('  PASS: 1080x1920 PNG without focus');

  // Test 3: Verify file size is reasonable (Sharp compression working)
  const stat = fs.statSync(result1.filepath);
  console.log(`  File size: ${(stat.size / 1024).toFixed(1)}KB`);
  if (stat.size > 500 * 1024) throw new Error('File too large — Sharp compression may not be working');
  if (stat.size < 1024) throw new Error('File too small — may be corrupt');
  console.log('  PASS: file size reasonable');

  // Cleanup test outputs
  fs.unlinkSync(result1.filepath);
  fs.unlinkSync(result2.filepath);

  console.log('\nAll render tests passed!');
}

runTest().catch(err => {
  console.error('FAIL:', err.message);
  process.exit(1);
});
```
